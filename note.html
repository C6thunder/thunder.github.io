// 导航栏隐藏/显示功能 document.addEventListener('DOMContentLoaded', function() { const sidebar = document.querySelector('.sidebar'); const mainContent = document.querySelector('.main-content'); const toggleBtn = document.querySelector('.sidebar-toggle-btn'); const showBtn = document.querySelector('.sidebar-show-btn'); if (!sidebar || !mainContent || !toggleBtn || !showBtn) return; // 恢复上次的状态 const sidebarHidden = localStorage.getItem('sidebarHidden') === 'true'; if (sidebarHidden) { sidebar.classList.add('hidden'); mainContent.classList.add('full-width'); showBtn.style.display = 'flex'; } // 隐藏按钮点击事件 toggleBtn.addEventListener('click', function() { sidebar.classList.add('hidden'); mainContent.classList.add('full-width'); showBtn.style.display = 'flex'; localStorage.setItem('sidebarHidden', 'true'); }); // 显示按钮点击事件 showBtn.addEventListener('click', function() { sidebar.classList.remove('hidden'); mainContent.classList.remove('full-width'); showBtn.style.display = 'none'; localStorage.setItem('sidebarHidden', 'false'); }); }); localStorage.setItem('sidebarHidden', 'true'); }); // 显示按钮点击事件 showBtn.addEventListener('click', function() { sidebar.classList.remove('hidden'); mainContent.classList.remove('full-width'); showBtn.style.display = 'none'; localStorage.setItem('sidebarHidden', 'false'); }); });
